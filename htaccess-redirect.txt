# Strato .htaccess - Multi-Domain Redirect
# Place this in /htdocs/.htaccess

RewriteEngine On

# Redirect dabrock.info to /dabrock-info/ subdirectory
RewriteCond %{HTTP_HOST} ^(www\.)?dabrock\.info$ [NC]
RewriteCond %{REQUEST_URI} !^/dabrock-info/
RewriteRule ^(.*)$ /dabrock-info/$1 [L]
